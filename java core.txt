12.if else(|| operator)
package Day1;
 
public class Program1 {
	
	public static void main(String[] args) {
		int a = 10;
		int b = 20;
		int c = 30;
		if(a>b || c>a) {
			System.out.println("a is small");
		}
		else {
			System.out.println("b is greater");
		}
	}
}
